> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `day_ex`, a standalone Elixir port of dayjs — a lightweight, pipe-friendly date/time library with formatting, comparison, relative time, timezone support, and duration arithmetic.

**Architecture:** Immutable `%DayEx{}` struct wrapping `DateTime.t()` or `NaiveDateTime.t()` with a locale atom. Pure functional — no GenServer, no global state. Recursive descent tokenizer for format strings. Locale data via behaviour modules. Duration stored as decomposed fields.

**Tech Stack:** Elixir 1.18+, OTP 28+, `tz_data` (timezone DB), `stream_data` (property tests)

  @doc "Current time in UTC."
  @spec now() :: t()
  def now, do: %DayEx{datetime: DateTime.utc_now()}

  @doc "Current time in UTC with locale."
  @spec now(atom()) :: t()
  def now(locale) when is_atom(locale), do: %DayEx{datetime: DateTime.utc_now(), locale: locale}

  @doc "Parse from various input types."
  @spec parse(term()) :: {:ok, t()} | {:error, String.t()}
  def parse(%DayEx{} = d), do: {:ok, %DayEx{datetime: d.datetime, locale: d.locale}}
  def parse(%DateTime{} = dt), do: {:ok, %DayEx{datetime: dt}}
  def parse(%NaiveDateTime{} = ndt), do: {:ok, %DayEx{datetime: ndt}}

  def parse(%Date{} = date) do
    {:ok, %DayEx{datetime: NaiveDateTime.new!(date, ~T[00:00:00])}}
  end

  def parse(ts) when is_integer(ts) do
    case DateTime.from_unix(ts) do
      {:ok, dt} -> {:ok, %DayEx{datetime: dt}}
      {:error, reason} -> {:error, "invalid unix timestamp: #{inspect(reason)}"}
    end
  end

  def parse(ts) when is_float(ts) do
    seconds = trunc(ts)
    microseconds = round((ts - seconds) * 1_000_000)

    case DateTime.from_unix(seconds) do
      {:ok, dt} ->
        dt = %{dt | microsecond: {microseconds, 6}}
        {:ok, %DayEx{datetime: dt}}

      {:error, reason} ->
        {:error, "invalid unix timestamp: #{inspect(reason)}"}
    end
  end

  def parse(str) when is_binary(str) do
    cond do
      String.contains?(str, "T") or String.contains?(str, "t") ->
        parse_iso8601_datetime(str)

      Regex.match?(~r/^\d{4}-\d{2}-\d{2}$/, str) ->
        case Date.from_iso8601(str) do
          {:ok, date} -> {:ok, %DayEx{datetime: NaiveDateTime.new!(date, ~T[00:00:00])}}
          {:error, reason} -> {:error, "invalid date: #{inspect(reason)}"}
        end

      true ->
        {:error, "unrecognized format: #{str}"}
    end
  end

  def parse(_), do: {:error, "unsupported input type"}

  defp parse_iso8601_datetime(str) do
    case DateTime.from_iso8601(str) do
      {:ok, dt, _offset} ->
        {:ok, %DayEx{datetime: dt}}

      {:error, _} ->
        case NaiveDateTime.from_iso8601(str) do
          {:ok, ndt} -> {:ok, %DayEx{datetime: ndt}}
          {:error, reason} -> {:error, "invalid datetime: #{inspect(reason)}"}
        end
    end
  end

  @doc "Parse from various input types. Raises on failure."
  @spec parse!(term()) :: t()
  def parse!(input) do
    case parse(input) do
      {:ok, d} -> d
      {:error, reason} -> raise ArgumentError, "failed to parse: #{reason}"
    end
  end

  @doc "Create from unix timestamp in seconds."
  @spec unix(integer()) :: t()
  def unix(ts) when is_integer(ts) do
    {:ok, dt} = DateTime.from_unix(ts)
    %DayEx{datetime: dt}
  end

  @doc "Current UTC time (explicit UTC mode)."
  @spec utc() :: t()
  def utc, do: %DayEx{datetime: DateTime.utc_now()}

  @doc "Parse input and convert to UTC."
  @spec utc(term()) :: t()
  def utc(input) do
    {:ok, d} = parse(input)

    case d.datetime do
      %DateTime{} = dt ->
        {:ok, utc_dt} = DateTime.shift_zone(dt, "Etc/UTC")
        %DayEx{datetime: utc_dt}

      %NaiveDateTime{} = ndt ->
        dt = DateTime.from_naive!(ndt, "Etc/UTC")
        %DayEx{datetime: dt}
    end
  end

  @doc "Set day of month."
  @spec date(t(), 1..31) :: t()
  def date(%DayEx{datetime: dt} = d, value) do
    %{d | datetime: update_datetime(dt, day: value)}
  end

  @doc "Set hour."
  @spec hour(t(), 0..23) :: t()
  def hour(%DayEx{datetime: dt} = d, value) do
    %{d | datetime: update_datetime(dt, hour: value)}
  end

  @doc "Set minute."
  @spec minute(t(), 0..59) :: t()
  def minute(%DayEx{datetime: dt} = d, value) do
    %{d | datetime: update_datetime(dt, minute: value)}
  end

  @doc "Set second."
  @spec second(t(), 0..59) :: t()
  def second(%DayEx{datetime: dt} = d, value) do
    %{d | datetime: update_datetime(dt, second: value)}
  end

  @doc "Set millisecond."
  @spec millisecond(t(), 0..999) :: t()
  def millisecond(%DayEx{datetime: dt} = d, value) do
    microsecond = value * 1000
    %{d | datetime: update_datetime(dt, microsecond: {microsecond, 3})}
  end

  @doc "Generic setter. Units: :year, :month, :date, :day, :hour, :minute, :second, :millisecond."
  @spec set(t(), atom(), integer()) :: t()
  def set(d, :year, value), do: year(d, value)
  def set(d, :month, value), do: month(d, value)
  def set(d, :date, value), do: date(d, value)
  def set(d, :hour, value), do: hour(d, value)
  def set(d, :minute, value), do: minute(d, value)
  def set(d, :second, value), do: second(d, value)
  def set(d, :millisecond, value), do: millisecond(d, value)

  defp update_datetime(%DateTime{} = dt, updates) do
    {date_fields, time_fields} = split_date_time_fields(updates)
    date = update_date(dt, date_fields)
    time = update_time(dt, time_fields)

    case DateTime.new(date, time, dt.time_zone) do
      {:ok, new_dt} -> new_dt
      {:ambiguous, first, _second} -> first
      {:gap, _just_before, just_after} -> just_after
    end
  end

  defp update_datetime(%NaiveDateTime{} = ndt, updates) do
    {date_fields, time_fields} = split_date_time_fields(updates)
    date = update_date(ndt, date_fields)
    time = update_time(ndt, time_fields)
    NaiveDateTime.new!(date, time)
  end

  defp split_date_time_fields(updates) do
    date_keys = [:year, :month, :day]
    date_fields = Keyword.take(updates, date_keys)
    time_fields = Keyword.drop(updates, date_keys)
    {date_fields, time_fields}
  end

  defp update_date(dt, []), do: Date.new!(dt.year, dt.month, dt.day)

  defp update_date(dt, fields) do
    year = Keyword.get(fields, :year, dt.year)
    month = Keyword.get(fields, :month, dt.month)
    day = Keyword.get(fields, :day, dt.day)
    max_day = Calendar.ISO.days_in_month(year, month)
    Date.new!(year, month, min(day, max_day))
  end

  defp update_time(dt, []) do
    Time.new!(dt.hour, dt.minute, dt.second, dt.microsecond)
  end

  defp update_time(dt, fields) do
    hour = Keyword.get(fields, :hour, dt.hour)
    minute = Keyword.get(fields, :minute, dt.minute)
    second = Keyword.get(fields, :second, dt.second)
    microsecond = Keyword.get(fields, :microsecond, dt.microsecond)
    Time.new!(hour, minute, second, microsecond)

  # --- Manipulation ---

  @doc "Add n units to a DayEx."
  @spec add(t(), integer(), atom()) :: t()
  def add(%DayEx{} = d, n, :year) do
    new_year = year(d) + n
    max_day = Calendar.ISO.days_in_month(new_year, month(d))
    clamped_day = min(date(d), max_day)
    %{d | datetime: update_datetime(d.datetime, year: new_year, day: clamped_day)}
  end

  def add(%DayEx{} = d, n, :month) do
    total_months = (year(d) - 1) * 12 + (month(d) - 1) + n
    new_year = div(total_months, 12) + 1
    new_month = rem(total_months, 12) + 1
    max_day = Calendar.ISO.days_in_month(new_year, new_month)
    clamped_day = min(date(d), max_day)
    %{d | datetime: update_datetime(d.datetime, year: new_year, month: new_month, day: clamped_day)}
  end

  def add(%DayEx{} = d, n, :week), do: add(d, n * 7, :day)

  def add(%DayEx{datetime: %DateTime{} = dt} = d, n, :day) do

defmodule DayEx.Format do
  @moduledoc "Recursive descent tokenizer and renderer for dayjs format tokens."

  @tokens ~w(YYYY YY MMMM MMM MM M DDDD DD Do D dddd ddd dd d HH H hh h mm m ss s SSS A a ZZ Z X x Q kk k GGGG GG WW W ww wo w)

  @doc "Tokenize a format string into a list of {:token, t} and {:literal, t} tuples."
  @spec tokenize(String.t()) :: [{:token, String.t()} | {:literal, String.t()}]
  def tokenize(format) when is_binary(format) do
    format
    |> do_tokenize([])
    |> Enum.reverse()
  end

  defp do_tokenize("", acc), do: acc

  defp do_tokenize("[" <> rest, acc) do
    case String.split(rest, "]", parts: 2) do
      [escaped, remaining] -> do_tokenize(remaining, [{:literal, escaped} | acc])
      _ -> do_tokenize("", [{:literal, "[" <> rest} | acc])
    end
  end

  defp do_tokenize(str, acc) do
    case match_token(str) do
      {token, rest} ->
        do_tokenize(rest, [{:token, token} | acc])

      nil ->
        {ch, rest} = String.split_at(str, 1)

        case acc do
          [{:literal, prev} | tail] -> do_tokenize(rest, [{:literal, prev <> ch} | tail])
          _ -> do_tokenize(rest, [{:literal, ch} | acc])
        end
    end
  end

  defp match_token(str) do
    Enum.find_value(@tokens, fn token ->
      if String.starts_with?(str, token) do
        {token, String.slice(str, String.length(token)..-1//1)}
      end
    end)
  end

  @doc "Format a DayEx with a format string."
  @spec format(DayEx.t(), String.t()) :: String.t()
  def format(%DayEx{} = d, format_str) do
    format_str
    |> tokenize()
    |> Enum.map(fn
      {:literal, text} -> text
      {:token, token} -> render_token(d, token)
    end)
    |> Enum.join()
  end

  defp render_token(d, "YYYY"), do: d |> DayEx.year() |> Integer.to_string()
  defp render_token(d, "YY"), do: d |> DayEx.year() |> rem(100) |> Integer.to_string() |> String.pad_leading(2, "0")
  defp render_token(d, "M"), do: d |> DayEx.month() |> Integer.to_string()
  defp render_token(d, "MM"), do: d |> DayEx.month() |> Integer.to_string() |> String.pad_leading(2, "0")

  defp render_token(d, "MMM") do

    Enum.at(locale_mod.months_short(), DayEx.month(d) - 1)
  end

  defp render_token(d, "MMMM") do
    locale_mod = DayEx.Locale.get(d.locale)
    Enum.at(locale_mod.months_full(), DayEx.month(d) - 1)
  end

  defp render_token(d, "D"), do: d |> DayEx.date() |> Integer.to_string()
  defp render_token(d, "DD"), do: d |> DayEx.date() |> Integer.to_string() |> String.pad_leading(2, "0")

  defp render_token(d, "Do") do
    locale_mod = DayEx.Locale.get(d.locale)
    locale_mod.ordinal(DayEx.date(d))
  end

  defp render_token(d, "d"), do: d |> DayEx.day() |> Integer.to_string()

  defp render_token(d, "dd") do
    locale_mod = DayEx.Locale.get(d.locale)
    Enum.at(locale_mod.weekdays_min(), DayEx.day(d))
  end

  defp render_token(d, "ddd") do

    Enum.at(locale_mod.weekdays_short(), DayEx.day(d))
  end

  defp render_token(d, "dddd") do

  end

  defp render_token(d, "HH"), do: d |> DayEx.hour() |> Integer.to_string() |> String.pad_leading(2, "0")

  defp render_token(d, "h") do
    h = DayEx.hour(d)
    h12 = case rem(h, 12) do
      0 -> 12
      n -> n
    end
    Integer.to_string(h12)
  end

  defp render_token(d, "hh") do

      n -> n

    h12 |> Integer.to_string() |> String.pad_leading(2, "0")
  end

  end

  defp render_token(d, "kk") do

  defp render_token(d, "mm"), do: d |> DayEx.minute() |> Integer.to_string() |> String.pad_leading(2, "0")

  defp render_token(d, "s"), do: d |> DayEx.second() |> Integer.to_string()
  defp render_token(d, "ss"), do: d |> DayEx.second() |> Integer.to_string() |> String.pad_leading(2, "0")

  defp render_token(d, "SSS"), do: d |> DayEx.millisecond() |> Integer.to_string() |> String.pad_leading(3, "0")

  defp render_token(d, "A") do

    {am, pm} = locale_mod.meridiem_upper()

  end

  defp render_token(d, "a") do
    locale_mod = DayEx.Locale.get(d.locale)
    {am, pm} = locale_mod.meridiem_lower()
    if DayEx.hour(d) < 12, do: am, else: pm
  end

  defp render_token(d, "Z") do
    case d.datetime do
      %DateTime{utc_offset: utc_offset, std_offset: std_offset} ->
        format_offset(utc_offset + std_offset, ":")

      %NaiveDateTime{} ->
        "+00:00"
    end
  end

  defp render_token(d, "ZZ") do
    case d.datetime do
      %DateTime{utc_offset: utc_offset, std_offset: std_offset} ->
        format_offset(utc_offset + std_offset, "")

      %NaiveDateTime{} ->
        "+0000"
    end
  end

  defp render_token(d, "X") do
    case d.datetime do
      %DateTime{} = dt -> dt |> DateTime.to_unix() |> Integer.to_string()
      %NaiveDateTime{} = ndt -> ndt |> DateTime.from_naive!("Etc/UTC") |> DateTime.to_unix() |> Integer.to_string()
    end
  end

  defp render_token(d, "x") do
    case d.datetime do
      %DateTime{} = dt -> dt |> DateTime.to_unix(:millisecond) |> Integer.to_string()
      %NaiveDateTime{} = ndt -> ndt |> DateTime.from_naive!("Etc/UTC") |> DateTime.to_unix(:millisecond) |> Integer.to_string()
    end
  end

  defp render_token(d, "Q") do
    q = div(DayEx.month(d) - 1, 3) + 1
    Integer.to_string(q)
  end

  defp render_token(d, "W") do
    {_year, week} = iso_week(d.datetime)
    Integer.to_string(week)
  end

  defp render_token(d, "WW") do
    {_year, week} = iso_week(d.datetime)
    week |> Integer.to_string() |> String.pad_leading(2, "0")
  end

  defp render_token(d, "GGGG") do
    {year, _week} = iso_week(d.datetime)
    Integer.to_string(year)
  end

  defp render_token(d, "GG") do
    {year, _week} = iso_week(d.datetime)
    year |> rem(100) |> Integer.to_string() |> String.pad_leading(2, "0")
  end

  defp render_token(d, "w") do
    # Locale-aware week — for :en (week_start=0, Sunday), same as simple calculation
    day_of_year = day_of_year(d.datetime)
    week = div(day_of_year - 1, 7) + 1
    Integer.to_string(week)
  end

  defp render_token(d, "ww") do
    day_of_year = day_of_year(d.datetime)
    week = div(day_of_year - 1, 7) + 1
    week |> Integer.to_string() |> String.pad_leading(2, "0")
  end

  defp render_token(d, "wo") do
    day_of_year = day_of_year(d.datetime)
    week = div(day_of_year - 1, 7) + 1
    locale_mod = DayEx.Locale.get(d.locale)
    locale_mod.ordinal(week)
  end

  defp render_token(d, "DDDD") do
    day_of_year(d.datetime) |> Integer.to_string() |> String.pad_leading(3, "0")
  end

  defp format_offset(total_seconds, separator) do
    sign = if total_seconds >= 0, do: "+", else: "-"
    abs_seconds = abs(total_seconds)
    hours = div(abs_seconds, 3600)
    minutes = div(rem(abs_seconds, 3600), 60)
    h_str = hours |> Integer.to_string() |> String.pad_leading(2, "0")
    m_str = minutes |> Integer.to_string() |> String.pad_leading(2, "0")
    "#{sign}#{h_str}#{separator}#{m_str}"
  end

  defp iso_week(dt) do
    date = case dt do
      %DateTime{} -> DateTime.to_date(dt)
      %NaiveDateTime{} -> NaiveDateTime.to_date(dt)
    end
    :calendar.iso_week_number({date.year, date.month, date.day})
  end

  defp day_of_year(dt) do
    date = case dt do
      %DateTime{} -> DateTime.to_date(dt)
      %NaiveDateTime{} -> NaiveDateTime.to_date(dt)
    end
    Date.day_of_year(date)
  end

# apps/day_ex/lib/day_ex/parse.ex
defmodule DayEx.Parse do
  @moduledoc "Parse date strings using dayjs-compatible format tokens."

  @doc "Parse a string using a format template. Returns {:ok, DayEx.t()} or {:error, reason}."
  @spec parse(String.t(), String.t(), atom()) :: {:ok, DayEx.t()} | {:error, String.t()}
  def parse(input, format, locale \\ :en) do
    tokens = DayEx.Format.tokenize(format)

    case extract_fields(input, tokens, %{}) do
      {:ok, fields} -> build_datetime(fields, locale)
      {:error, reason} -> {:error, reason}
    end
  end

  defp extract_fields("", [], fields), do: {:ok, fields}
  defp extract_fields(_, [], _), do: {:error, "trailing input after format consumed"}
  defp extract_fields("", _, _), do: {:error, "input exhausted before format consumed"}

  defp extract_fields(input, [{:literal, lit} | rest], fields) do
    lit_len = String.length(lit)

    if String.starts_with?(input, lit) do
      extract_fields(String.slice(input, lit_len..-1//1), rest, fields)
    else
      {:error, "expected literal '#{lit}' but got '#{String.slice(input, 0, lit_len)}'"}
    end
  end

  defp extract_fields(input, [{:token, token} | rest], fields) do
    case consume_token(input, token, fields) do
      {:ok, value, remaining, field_key} ->
        extract_fields(remaining, rest, Map.put(fields, field_key, value))

      {:error, reason} ->
        {:error, reason}
    end
  end

  defp consume_token(input, "YYYY", _fields), do: consume_digits(input, 4, :year)
  defp consume_token(input, "YY", _fields), do: consume_digits_then(input, 2, :year, fn v -> v + 2000 end)
  defp consume_token(input, "MM", _fields), do: consume_digits(input, 2, :month)
  defp consume_token(input, "M", _fields), do: consume_digits_variable(input, 1, 2, :month)
  defp consume_token(input, "DD", _fields), do: consume_digits(input, 2, :day)
  defp consume_token(input, "D", _fields), do: consume_digits_variable(input, 1, 2, :day)
  defp consume_token(input, "HH", _fields), do: consume_digits(input, 2, :hour)
  defp consume_token(input, "H", _fields), do: consume_digits_variable(input, 1, 2, :hour)
  defp consume_token(input, "hh", _fields), do: consume_digits(input, 2, :hour12)
  defp consume_token(input, "h", _fields), do: consume_digits_variable(input, 1, 2, :hour12)
  defp consume_token(input, "mm", _fields), do: consume_digits(input, 2, :minute)
  defp consume_token(input, "m", _fields), do: consume_digits_variable(input, 1, 2, :minute)
  defp consume_token(input, "ss", _fields), do: consume_digits(input, 2, :second)
  defp consume_token(input, "s", _fields), do: consume_digits_variable(input, 1, 2, :second)
  defp consume_token(input, "SSS", _fields), do: consume_digits(input, 3, :millisecond)

  defp consume_token(input, "A", _fields) do
    cond do
      String.starts_with?(input, "AM") -> {:ok, :am, String.slice(input, 2..-1//1), :meridiem}
      String.starts_with?(input, "PM") -> {:ok, :pm, String.slice(input, 2..-1//1), :meridiem}
      true -> {:error, "expected AM/PM"}
    end
  end

  defp consume_token(input, "a", _fields) do
    cond do
      String.starts_with?(input, "am") -> {:ok, :am, String.slice(input, 2..-1//1), :meridiem}
      String.starts_with?(input, "pm") -> {:ok, :pm, String.slice(input, 2..-1//1), :meridiem}
      true -> {:error, "expected am/pm"}
    end
  end

  defp consume_token(input, "MMM", fields) do
    locale = Map.get(fields, :locale, :en)
    locale_mod = DayEx.Locale.get(locale)
    months = locale_mod.months_short()

    Enum.find_value(Enum.with_index(months, 1), {:error, "expected short month name"}, fn {name, idx} ->
      if String.starts_with?(input, name) do
        {:ok, idx, String.slice(input, String.length(name)..-1//1), :month}
      end
    end)
  end

  defp consume_token(input, "MMMM", fields) do
    locale = Map.get(fields, :locale, :en)
    locale_mod = DayEx.Locale.get(locale)
    months = locale_mod.months_full()

    Enum.find_value(Enum.with_index(months, 1), {:error, "expected full month name"}, fn {name, idx} ->
      if String.starts_with?(input, name) do
        {:ok, idx, String.slice(input, String.length(name)..-1//1), :month}
      end
    end)
  end

  defp consume_token(input, _token, _fields) do
    # Skip unknown tokens by consuming no input
    {:ok, nil, input, :_skip}
  end

  defp consume_digits(input, count, field) do
    {digits, rest} = String.split_at(input, count)

    case Integer.parse(digits) do
      {value, ""} -> {:ok, value, rest, field}
      _ -> {:error, "expected #{count} digits for #{field}, got '#{digits}'"}
    end
  end

  defp consume_digits_variable(input, min, max, field) do
    {candidate, _} = String.split_at(input, max)

    # Try max first, then min
    case Integer.parse(candidate) do
      {value, ""} -> {:ok, value, String.slice(input, max..-1//1), field}
      _ ->
        {candidate_min, _} = String.split_at(input, min)
        case Integer.parse(candidate_min) do
          {value, ""} -> {:ok, value, String.slice(input, min..-1//1), field}
          _ -> {:error, "expected #{min}-#{max} digits for #{field}"}
        end
    end
  end

  defp consume_digits_then(input, count, field, transform) do
    case consume_digits(input, count, field) do
      {:ok, value, rest, field} -> {:ok, transform.(value), rest, field}
      error -> error
    end
  end

  defp build_datetime(fields, locale) do
    year = Map.get(fields, :year, 2000)
    month = Map.get(fields, :month, 1)
    day = Map.get(fields, :day, 1)
    minute = Map.get(fields, :minute, 0)
    second = Map.get(fields, :second, 0)
    millisecond = Map.get(fields, :millisecond, 0)

    hour =
      case {Map.get(fields, :hour), Map.get(fields, :hour12), Map.get(fields, :meridiem)} do
        {h, nil, _} when is_integer(h) -> h
        {nil, h12, :pm} when h12 < 12 -> h12 + 12
        {nil, 12, :am} -> 0
        {nil, h12, _} when is_integer(h12) -> h12
        _ -> Map.get(fields, :hour, 0)
      end

    microsecond = {millisecond * 1000, 3}

    case NaiveDateTime.new(year, month, day, hour, minute, second, microsecond) do
      {:ok, ndt} -> {:ok, %DayEx{datetime: ndt, locale: locale}}
      {:error, reason} -> {:error, "invalid date: #{inspect(reason)}"}
    end
  end
end

# apps/day_ex/lib/day_ex/relative_time.ex
defmodule DayEx.RelativeTime do
  @moduledoc "Relative time formatting with dayjs-compatible thresholds."

  @doc "Relative time string from `d` compared to `reference`."
  @spec from(DayEx.t(), DayEx.t(), boolean()) :: String.t()
  def from(%DayEx{} = d, %DayEx{} = reference, without_suffix \\ false) do
    diff_ms = DayEx.diff(d, reference)
    abs_ms = abs(diff_ms)
    future? = diff_ms > 0

    locale_mod = DayEx.Locale.get(d.locale)
    rt = locale_mod.relative_time()

    raw = relative_string(abs_ms, rt)

    if without_suffix do
      raw
    else
      if future? do
        "in #{raw}"
      else
        "#{raw} ago"
      end
    end
  end

  @doc "Relative time string from `d` to `reference`."
  @spec to(DayEx.t(), DayEx.t(), boolean()) :: String.t()
  def to(%DayEx{} = d, %DayEx{} = reference, without_suffix \\ false) do
    from(reference, d, without_suffix)
  end

  defp relative_string(abs_ms, rt) do
    seconds = div(abs_ms, 1_000)
    minutes = div(abs_ms, 60_000)
    hours = div(abs_ms, 3_600_000)
    days = div(abs_ms, 86_400_000)
    months = div(abs_ms, 2_629_746_000)

    cond do
      seconds < 45 -> rt.s
      seconds < 90 -> rt.m
      minutes < 45 -> String.replace(rt.mm, "%d", Integer.to_string(minutes))
      minutes < 90 -> rt.h
      hours < 22 -> String.replace(rt.hh, "%d", Integer.to_string(hours))
      hours < 36 -> rt.d
      days < 26 -> String.replace(rt.dd, "%d", Integer.to_string(days))
      days < 46 -> rt.M
      months < 11 -> String.replace(rt.MM, "%d", Integer.to_string(months))
      months < 18 -> rt.y
      true -> String.replace(rt.yy, "%d", Integer.to_string(months |> div(12)))
    end
  end
end

  # --- Timezone ---

  @doc "Parse a string in a specific timezone."
  @spec tz(String.t(), String.t()) :: t()
  def tz(input, timezone) when is_binary(input) and is_binary(timezone) do
    {:ok, d} = parse(input)

    case d.datetime do
      %NaiveDateTime{} = ndt ->
        case DateTime.from_naive(ndt, timezone) do
          {:ok, dt} -> %{d | datetime: dt}
          {:ambiguous, first, _} -> %{d | datetime: first}
          {:gap, _, just_after} -> %{d | datetime: just_after}
        end

      %DateTime{} = dt ->
        {:ok, shifted} = DateTime.shift_zone(dt, timezone)
        %{d | datetime: shifted}
    end
  end

  @doc "Convert a DayEx to a different timezone."
  @spec tz(t(), String.t()) :: t()
  def tz(%DayEx{datetime: %DateTime{} = dt} = d, timezone) when is_binary(timezone) do
    {:ok, shifted} = DateTime.shift_zone(dt, timezone)
    %{d | datetime: shifted}
  end

  def tz(%DayEx{datetime: %NaiveDateTime{} = ndt} = d, timezone) when is_binary(timezone) do
    dt = DateTime.from_naive!(ndt, "Etc/UTC")
    {:ok, shifted} = DateTime.shift_zone(dt, timezone)
    %{d | datetime: shifted}
  end

  @doc "Get the timezone name."
  @spec tz_name(t()) :: String.t() | nil
  def tz_name(%DayEx{datetime: %DateTime{time_zone: tz}}), do: tz
  def tz_name(%DayEx{datetime: %NaiveDateTime{}}), do: nil

  @doc "Convert to local (NaiveDateTime)."
  @spec local(t()) :: t()
  def local(%DayEx{datetime: %DateTime{} = dt} = d) do
    %{d | datetime: DateTime.to_naive(dt)}
  end

  def local(%DayEx{datetime: %NaiveDateTime{}} = d), do: d

  @doc "Get UTC offset in minutes."
  @spec utc_offset(t()) :: integer() | nil
  def utc_offset(%DayEx{datetime: %DateTime{utc_offset: offset, std_offset: std}}) do

# apps/day_ex/lib/day_ex/duration.ex
defmodule DayEx.Duration do
  @moduledoc "Duration type with humanize support and ISO 8601 formatting."

  @type t :: %__MODULE__{
          years: integer(),
          months: integer(),
          days: integer(),
          hours: integer(),
          minutes: integer(),
          seconds: integer(),
          milliseconds: integer()
        }

  defstruct years: 0, months: 0, days: 0, hours: 0, minutes: 0, seconds: 0, milliseconds: 0

  @ms_per_second 1_000
  @ms_per_minute 60_000
  @ms_per_hour 3_600_000
  @ms_per_day 86_400_000

  @doc "Create a Duration from milliseconds, a map, or an ISO 8601 duration string."
  @spec new(integer() | map() | String.t()) :: t()
  def new(ms) when is_integer(ms) do
    abs_ms = abs(ms)
    sign = if ms < 0, do: -1, else: 1

    hours = div(abs_ms, @ms_per_hour)
    remaining = rem(abs_ms, @ms_per_hour)
    minutes = div(remaining, @ms_per_minute)
    remaining = rem(remaining, @ms_per_minute)
    seconds = div(remaining, @ms_per_second)
    milliseconds = rem(remaining, @ms_per_second)

    %__MODULE__{
      hours: sign * hours,
      minutes: sign * minutes,
      seconds: sign * seconds,
      milliseconds: sign * milliseconds
    }
  end

  def new(%{} = map) do
    %__MODULE__{
      years: Map.get(map, :years, 0),
      months: Map.get(map, :months, 0),
      days: Map.get(map, :days, 0),
      hours: Map.get(map, :hours, 0),
      minutes: Map.get(map, :minutes, 0),
      seconds: Map.get(map, :seconds, 0),
      milliseconds: Map.get(map, :milliseconds, 0)
    }
  end

  def new(str) when is_binary(str) do
    parse_iso8601(str)
  end

  @doc "Get years component."
  @spec years(t()) :: integer()
  def years(%__MODULE__{years: v}), do: v

  @doc "Get months component."
  @spec months(t()) :: integer()
  def months(%__MODULE__{months: v}), do: v

  @doc "Get days component."
  @spec days(t()) :: integer()
  def days(%__MODULE__{days: v}), do: v

  @doc "Get hours component."
  @spec hours(t()) :: integer()
  def hours(%__MODULE__{hours: v}), do: v

  @doc "Get minutes component."
  @spec minutes(t()) :: integer()
  def minutes(%__MODULE__{minutes: v}), do: v

  @doc "Get seconds component."
  @spec seconds(t()) :: integer()
  def seconds(%__MODULE__{seconds: v}), do: v

  @doc "Get milliseconds component."
  @spec milliseconds(t()) :: integer()
  def milliseconds(%__MODULE__{milliseconds: v}), do: v

  @doc "Total as milliseconds."
  @spec as_milliseconds(t()) :: integer()
  def as_milliseconds(%__MODULE__{} = d) do
    total_ms(d)
  end

  @doc "Total as seconds (float)."
  @spec as_seconds(t()) :: float()
  def as_seconds(%__MODULE__{} = d), do: total_ms(d) / @ms_per_second

  @doc "Total as minutes (float)."
  @spec as_minutes(t()) :: float()
  def as_minutes(%__MODULE__{} = d), do: total_ms(d) / @ms_per_minute

  @doc "Total as hours (float)."
  @spec as_hours(t()) :: float()
  def as_hours(%__MODULE__{} = d), do: total_ms(d) / @ms_per_hour

  @doc "Total as days (float)."
  @spec as_days(t()) :: float()
  def as_days(%__MODULE__{} = d), do: total_ms(d) / @ms_per_day

  @doc "Total as weeks (float)."
  @spec as_weeks(t()) :: float()
  def as_weeks(%__MODULE__{} = d), do: total_ms(d) / (@ms_per_day * 7)

  @doc "Total as months (approximate, float)."
  @spec as_months(t()) :: float()
  def as_months(%__MODULE__{} = d), do: total_ms(d) / 2_629_746_000

  @doc "Total as years (approximate, float)."
  @spec as_years(t()) :: float()
  def as_years(%__MODULE__{} = d), do: total_ms(d) / 31_556_952_000

  @doc "Human-readable duration string."
  @spec humanize(t()) :: String.t()
  def humanize(%__MODULE__{} = d) do
    ms = abs(total_ms(d))
    seconds = div(ms, 1_000)
    minutes = div(ms, 60_000)
    hours = div(ms, 3_600_000)
    days = div(ms, 86_400_000)
    months = div(ms, 2_629_746_000)

    cond do
      seconds < 45 -> "a few seconds"
      seconds < 90 -> "a minute"
      minutes < 45 -> "#{minutes} minutes"
      minutes < 90 -> "an hour"
      hours < 22 -> "#{hours} hours"
      hours < 36 -> "a day"
      days < 26 -> "#{days} days"
      days < 46 -> "a month"
      months < 11 -> "#{months} months"
      months < 18 -> "a year"
      true -> "#{div(months, 12)} years"
    end
  end

  @doc "Human-readable duration string with suffix."
  @spec humanize(t(), boolean()) :: String.t()
  def humanize(%__MODULE__{} = d, true) do
    raw = humanize(d)
    ms = total_ms(d)

    if ms >= 0 do
      "in #{raw}"
    else
      "#{raw} ago"
    end
  end

  def humanize(%__MODULE__{} = d, false), do: humanize(d)

  @doc "Convert to ISO 8601 duration string."
  @spec to_iso_string(t()) :: String.t()
  def to_iso_string(%__MODULE__{} = d) do
    date_parts =
      [
        if(d.years != 0, do: "#{d.years}Y"),
        if(d.months != 0, do: "#{d.months}M"),
        if(d.days != 0, do: "#{d.days}D")
      ]
      |> Enum.reject(&is_nil/1)
      |> Enum.join()

    time_parts =
      [
        if(d.hours != 0, do: "#{d.hours}H"),
        if(d.minutes != 0, do: "#{d.minutes}M"),
        if(d.seconds != 0 or d.milliseconds != 0,
          do:
            if(d.milliseconds != 0,
              do: "#{d.seconds}.#{d.milliseconds |> Integer.to_string() |> String.pad_leading(3, "0")}S",
              else: "#{d.seconds}S"
            ))
      ]
      |> Enum.reject(&is_nil/1)
      |> Enum.join()

    time_str = if time_parts != "", do: "T#{time_parts}", else: ""

    case "P#{date_parts}#{time_str}" do
      "P" -> "P0D"
      result -> result
    end
  end

  @doc "Add two durations."
  @spec add(t(), t()) :: t()
  def add(%__MODULE__{} = a, %__MODULE__{} = b) do
    %__MODULE__{
      years: a.years + b.years,
      months: a.months + b.months,
      days: a.days + b.days,
      hours: a.hours + b.hours,
      minutes: a.minutes + b.minutes,
      seconds: a.seconds + b.seconds,
      milliseconds: a.milliseconds + b.milliseconds
    }
  end

  @doc "Subtract duration b from a."
  @spec subtract(t(), t()) :: t()
  def subtract(%__MODULE__{} = a, %__MODULE__{} = b) do
    %__MODULE__{
      years: a.years - b.years,
      months: a.months - b.months,
      days: a.days - b.days,
      hours: a.hours - b.hours,
      minutes: a.minutes - b.minutes,
      seconds: a.seconds - b.seconds,
      milliseconds: a.milliseconds - b.milliseconds
    }
  end

  defp total_ms(%__MODULE__{} = d) do
    d.years * 31_556_952_000 +
      d.months * 2_629_746_000 +
      d.days * @ms_per_day +
      d.hours * @ms_per_hour +
      d.minutes * @ms_per_minute +
      d.seconds * @ms_per_second +
      d.milliseconds
  end

  defp parse_iso8601("P" <> rest) do
    {date_part, time_part} =
      case String.split(rest, "T", parts: 2) do
        [date, time] -> {date, time}
        [date] -> {date, ""}
      end

    date_fields = parse_iso_date_part(date_part)
    time_fields = parse_iso_time_part(time_part)

    new(Map.merge(date_fields, time_fields))
  end

  defp parse_iso_date_part(""), do: %{}

  defp parse_iso_date_part(str) do
    Regex.scan(~r/(\d+)([YMD])/, str)
    |> Enum.reduce(%{}, fn [_, n, unit], acc ->
      value = String.to_integer(n)

      case unit do
        "Y" -> Map.put(acc, :years, value)
        "M" -> Map.put(acc, :months, value)
        "D" -> Map.put(acc, :days, value)
      end
    end)
  end

  defp parse_iso_time_part(""), do: %{}

  defp parse_iso_time_part(str) do
    Regex.scan(~r/(\d+(?:\.\d+)?)([HMS])/, str)
    |> Enum.reduce(%{}, fn [_, n, unit], acc ->
      case unit do
        "H" -> Map.put(acc, :hours, String.to_integer(n))
        "M" -> Map.put(acc, :minutes, String.to_integer(n))
        "S" ->
          if String.contains?(n, ".") do
            {secs, "." <> frac} = Float.parse(n) |> then(fn {f, _} -> {trunc(f), "." <> String.split(n, ".") |> List.last()} end)
            ms = String.pad_trailing(frac |> String.slice(1..-1//1), 3, "0") |> String.slice(0, 3) |> String.to_integer()
            acc |> Map.put(:seconds, secs) |> Map.put(:milliseconds, ms)
          else
            Map.put(acc, :seconds, String.to_integer(n))
          end
      end
    end)
  end
end
