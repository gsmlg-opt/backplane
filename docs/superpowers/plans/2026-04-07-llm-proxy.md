# LLM Proxy Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a sixth engine to Backplane — an LLM API proxy that routes requests to upstream LLM providers (Anthropic/OpenAI format), injects credentials, streams responses via Relayixir, and tracks usage.

**Architecture:** Clients send LLM requests to `/llm/v1/messages` (Anthropic) or `/llm/v1/chat/completions` (OpenAI). A Plug.Router extracts the model field, resolves the provider via prefixed names or global aliases, rewrites the model field, injects credentials, and delegates to Relayixir for streaming HTTP proxy. Usage is collected via Telemetry → Oban → PostgreSQL. Admin API and LiveView for provider CRUD.

**Tech Stack:** Elixir/OTP 28+, Ecto/PostgreSQL, Oban, Relayixir (in-umbrella HTTP proxy), Phoenix LiveView, AES-256-GCM encryption via `:crypto`, ETS for rate limiting and caching.

```elixir
# apps/backplane/lib/backplane/llm/encryption.ex
defmodule Backplane.LLM.Encryption do
  @moduledoc """
  AES-256-GCM encryption for LLM provider API keys.
  """

  @aad "backplane_llm_api_key"

  @doc "Derive a 32-byte key from a secret_key_base string."
  @spec derive_key(String.t()) :: binary()
  def derive_key(secret) when byte_size(secret) >= 32 do
    :crypto.hash(:sha256, secret)
  end

  @doc "Encrypt plaintext using AES-256-GCM with a random 12-byte IV."
  @spec encrypt(String.t(), binary()) :: binary()
  def encrypt(plaintext, key) when is_binary(plaintext) and byte_size(key) == 32 do
    iv = :crypto.strong_rand_bytes(12)

    {ciphertext, tag} =
      :crypto.crypto_one_time_aead(:aes_256_gcm, key, iv, plaintext, @aad, true)

    iv <> tag <> ciphertext
  end

  @doc "Decrypt a blob encrypted by encrypt/2. Returns {:ok, plaintext} or :error."
  @spec decrypt(binary(), binary()) :: {:ok, String.t()} | :error
  def decrypt(<<iv::binary-12, tag::binary-16, ciphertext::binary>>, key)
      when byte_size(key) == 32 do
    case :crypto.crypto_one_time_aead(:aes_256_gcm, key, iv, ciphertext, @aad, tag, false) do
      plaintext when is_binary(plaintext) -> {:ok, plaintext}
      :error -> :error
    end
  end

  def decrypt(_, _), do: :error

  @doc "Get the encryption key from application config."
  @spec get_key() :: binary()
  def get_key do
    secret =
      Application.get_env(:backplane, :secret_key_base) ||
        Application.get_env(:backplane_web, BackplaneWeb.Endpoint)[:secret_key_base] ||
        raise "No secret_key_base configured for LLM API key encryption"

    derive_key(secret)
  end
end
```

```elixir
# apps/backplane/lib/backplane/llm/provider.ex
defmodule Backplane.LLM.Provider do
  @moduledoc """
  Ecto schema and context for LLM providers.
  Handles CRUD, API key encryption, and soft delete.
  """

  use Ecto.Schema
  import Ecto.Changeset
  import Ecto.Query

  alias Backplane.LLM.{Encryption, ModelAlias}
  alias Backplane.Repo

  @type t :: %__MODULE__{}
  @primary_key {:id, :binary_id, autogenerate: true}
  @timestamps_opts [type: :utc_datetime_usec]

  @name_format ~r/^[a-z0-9][a-z0-9-]*$/

  schema "llm_providers" do
    field :name, :string
    field :api_type, Ecto.Enum, values: [:anthropic, :openai]
    field :api_url, :string
    field :api_key_encrypted, :binary
    field :api_key, :string, virtual: true
    field :models, {:array, :string}
    field :default_headers, :map, default: %{}
    field :rpm_limit, :integer
    field :enabled, :boolean, default: true
    field :deleted_at, :utc_datetime_usec

    has_many :aliases, ModelAlias
    has_many :usage_logs, Backplane.LLM.UsageLog

    timestamps()
  end

  @required_fields ~w(name api_type api_url models)a
  @optional_fields ~w(default_headers rpm_limit enabled)a

  @doc "Create changeset for a new provider. Requires `api_key` virtual field."
  def changeset(provider, attrs) do
    provider
    |> cast(attrs, @required_fields ++ @optional_fields ++ [:api_key])
    |> validate_required(@required_fields)
    |> validate_required_api_key()
    |> validate_format(:name, @name_format, message: "must be lowercase alphanumeric with hyphens, starting with alphanumeric")
    |> validate_api_url()
    |> validate_length(:models, min: 1, message: "must have at least one model")
    |> unsafe_validate_unique(:name, Repo, query: from(p in __MODULE__, where: is_nil(p.deleted_at)))
    |> unique_constraint(:name, name: :llm_providers_name_active_index)
    |> encrypt_api_key()
  end

  @doc "Update changeset. `api_key` is optional — omit to keep existing."
  def update_changeset(provider, attrs) do
    provider
    |> cast(attrs, @optional_fields ++ [:api_key, :models, :api_url])
    |> validate_api_url()
    |> encrypt_api_key()
  end

  defp validate_required_api_key(changeset) do
    # Only require api_key on insert (new record with no existing encrypted key)
    if get_field(changeset, :api_key_encrypted) == nil and get_change(changeset, :api_key) == nil do
      add_error(changeset, :api_key, "can't be blank")
    else
      changeset
    end
  end

  defp validate_api_url(changeset) do
    case get_field(changeset, :api_url) do
      nil ->
        changeset

      url ->
        uri = URI.parse(url)

        cond do
          uri.scheme == "https" ->
            changeset

          uri.scheme == "http" and uri.host in ["localhost", "127.0.0.1"] ->
            changeset

          uri.scheme == "http" ->
            add_error(changeset, :api_url, "must use https:// (http:// only allowed for localhost)")

          true ->
            add_error(changeset, :api_url, "must start with https://")
        end
    end
  end

  defp encrypt_api_key(changeset) do
    case get_change(changeset, :api_key) do
      nil ->
        changeset

      plaintext ->
        key = Encryption.get_key()
        encrypted = Encryption.encrypt(plaintext, key)

        changeset
        |> put_change(:api_key_encrypted, encrypted)
        |> delete_change(:api_key)
    end
  end

  # --- Context Functions ---

  @doc "Create a new provider."
  @spec create(map()) :: {:ok, t()} | {:error, Ecto.Changeset.t()}
  def create(attrs) do
    result =
      %__MODULE__{}
      |> changeset(attrs)
      |> Repo.insert()

    if match?({:ok, _}, result), do: broadcast_change()
    result
  end

  @doc "Update an existing provider."
  @spec update(t(), map()) :: {:ok, t()} | {:error, Ecto.Changeset.t()}
  def update(%__MODULE__{} = provider, attrs) do
    result =
      provider
      |> update_changeset(attrs)
      |> Repo.update()

    if match?({:ok, _}, result), do: broadcast_change()
    result
  end

  @doc "Soft-delete a provider: sets deleted_at, disables, removes aliases."
  @spec soft_delete(t()) :: {:ok, t()} | {:error, Ecto.Changeset.t()}
  def soft_delete(%__MODULE__{} = provider) do
    Repo.transaction(fn ->
      # Hard-delete aliases
      from(a in ModelAlias, where: a.provider_id == ^provider.id) |> Repo.delete_all()

      # Soft-delete the provider
      provider
      |> change(%{deleted_at: DateTime.utc_now(), enabled: false})
      |> Repo.update!()
    end)
    |> case do
      {:ok, provider} ->
        broadcast_change()
        {:ok, provider}

      error ->
        error
    end
  end

  @doc "List all active (non-deleted) providers."
  @spec list() :: [t()]
  def list do
    __MODULE__
    |> where([p], is_nil(p.deleted_at))
    |> order_by(:name)
    |> Repo.all()
    |> Repo.preload(:aliases)
  end

  @doc "Get a provider by ID (only active)."
  @spec get(String.t()) :: t() | nil
  def get(id) do
    __MODULE__
    |> where([p], p.id == ^id and is_nil(p.deleted_at))
    |> Repo.one()
    |> case do
      nil -> nil
      provider -> Repo.preload(provider, :aliases)
    end
  end

  @doc "Decrypt a provider's API key."
  @spec decrypt_api_key(t()) :: {:ok, String.t()} | :error
  def decrypt_api_key(%__MODULE__{api_key_encrypted: encrypted}) do
    Encryption.decrypt(encrypted, Encryption.get_key())
```

```elixir
# apps/backplane/lib/backplane/llm/model_alias.ex
defmodule Backplane.LLM.ModelAlias do
  @moduledoc """
  Ecto schema and context for global LLM model aliases.
  An alias maps an unprefixed name to a specific {provider, model} pair.
  """

  use Ecto.Schema
  import Ecto.Changeset
  import Ecto.Query

  alias Backplane.LLM.Provider
  alias Backplane.Repo

  @type t :: %__MODULE__{}
  @primary_key {:id, :binary_id, autogenerate: true}
  @timestamps_opts [type: :utc_datetime_usec]

  schema "llm_model_aliases" do
    field :alias, :string
    field :model, :string
    belongs_to :provider, Provider, type: :binary_id

    timestamps()
  end

  def changeset(alias_record, attrs) do
    alias_record
    |> cast(attrs, [:alias, :model, :provider_id])
    |> validate_required([:alias, :model, :provider_id])
    |> validate_format(:alias, ~r/^[^\/]+$/, message: "must not contain slashes")
    |> validate_model_in_provider()
    |> validate_provider_active()
    |> unique_constraint(:alias)
  end

  defp validate_model_in_provider(changeset) do
    provider_id = get_field(changeset, :provider_id)
    model = get_field(changeset, :model)

    if provider_id && model do
      case Repo.get(Provider, provider_id) do
        nil ->
          add_error(changeset, :provider_id, "provider not found")

        provider ->
          if model in provider.models do
            changeset
          else
            add_error(changeset, :model, "must be one of the provider's models: #{Enum.join(provider.models, ", ")}")
          end
      end
    else
      changeset
    end
  end

  defp validate_provider_active(changeset) do
    provider_id = get_field(changeset, :provider_id)

    if provider_id do
      case Repo.get(Provider, provider_id) do
        nil ->
          add_error(changeset, :provider_id, "provider not found")

        %{deleted_at: deleted_at} when not is_nil(deleted_at) ->
          add_error(changeset, :provider_id, "provider has been deleted")

        _ ->
          changeset
      end
    else
      changeset
    end
  end

  # --- Context Functions ---

  @spec create(map()) :: {:ok, t()} | {:error, Ecto.Changeset.t()}
  def create(attrs) do
    result =
      %__MODULE__{}
      |> changeset(attrs)
      |> Repo.insert()

    if match?({:ok, _}, result), do: broadcast_change()
    result
  end

  @spec delete(t()) :: {:ok, t()} | {:error, Ecto.Changeset.t()}
  def delete(%__MODULE__{} = alias_record) do
    result = Repo.delete(alias_record)
    if match?({:ok, _}, result), do: broadcast_change()
    result
  end

  @spec list() :: [t()]
  def list do
    __MODULE__
    |> order_by(:alias)
    |> Repo.all()
    |> Repo.preload(:provider)
  end

  @spec get(String.t()) :: t() | nil
  def get(id), do: Repo.get(__MODULE__, id) |> Repo.preload(:provider)

  @doc "Resolve an alias to {provider, raw_model}. Only returns active, non-deleted providers."
  @spec resolve(String.t()) :: {:ok, Provider.t(), String.t()} | {:error, :not_found}
  def resolve(alias_name) do
    query =
      from a in __MODULE__,
        join: p in Provider,
        on: a.provider_id == p.id,
        where: a.alias == ^alias_name and p.enabled == true and is_nil(p.deleted_at),
        select: {p, a.model}

    case Repo.one(query) do
      {provider, model} -> {:ok, provider, model}
      nil -> {:error, :not_found}
    end
  end

  defp broadcast_change do
    Phoenix.PubSub.broadcast(Backplane.PubSub, "llm:providers", {:llm_providers_changed, %{}})
  end
end
```

```elixir
# apps/backplane/lib/backplane/llm/api_router.ex
defmodule Backplane.LLM.ApiRouter do
  @moduledoc """
  REST API router for LLM provider and alias management.
  Mounted at /api/llm in the Phoenix router.
  """

  use Plug.Router

  alias Backplane.LLM.{Provider, ModelAlias}

  plug :match

  plug Plug.Parsers,
    parsers: [:json],
    pass: ["application/json"],
    json_decoder: Jason

  plug :dispatch

  # --- Providers ---

  get "/providers" do
    providers = Provider.list()
    send_json(conn, 200, Enum.map(providers, &serialize_provider/1))
  end

  get "/providers/:id" do
    case Provider.get(id) do
      nil -> send_json(conn, 404, %{error: "Provider not found"})
      provider -> send_json(conn, 200, serialize_provider(provider))
    end
  end

  post "/providers" do
    attrs = atomize_keys(conn.body_params)

    case Provider.create(attrs) do
      {:ok, provider} ->
        provider = Backplane.Repo.preload(provider, :aliases)
        send_json(conn, 201, serialize_provider(provider))

      {:error, changeset} ->
        send_json(conn, 422, %{errors: format_errors(changeset)})
    end
  end

  patch "/providers/:id" do
    case Provider.get(id) do
      nil ->
        send_json(conn, 404, %{error: "Provider not found"})

      provider ->
        attrs = atomize_keys(conn.body_params)

        case Provider.update(provider, attrs) do
          {:ok, updated} ->
            updated = Backplane.Repo.preload(updated, :aliases)
            send_json(conn, 200, serialize_provider(updated))

          {:error, changeset} ->
            send_json(conn, 422, %{errors: format_errors(changeset)})
        end
    end
  end

  delete "/providers/:id" do
    case Provider.get(id) do
      nil ->
        send_json(conn, 404, %{error: "Provider not found"})

      provider ->
        case Provider.soft_delete(provider) do
          {:ok, _} -> send_json(conn, 200, %{ok: true})
          {:error, _} -> send_json(conn, 500, %{error: "Failed to delete"})
        end
    end
  end

  # --- Aliases ---

  get "/aliases" do
    aliases = ModelAlias.list()
    send_json(conn, 200, Enum.map(aliases, &serialize_alias/1))
  end

  post "/aliases" do
    attrs = atomize_keys(conn.body_params)

    case ModelAlias.create(attrs) do
      {:ok, alias_record} ->
        alias_record = Backplane.Repo.preload(alias_record, :provider)
        send_json(conn, 201, serialize_alias(alias_record))

      {:error, changeset} ->
        send_json(conn, 422, %{errors: format_errors(changeset)})
    end
  end

  delete "/aliases/:id" do
    case ModelAlias.get(id) do
      nil ->
        send_json(conn, 404, %{error: "Alias not found"})

      alias_record ->
        case ModelAlias.delete(alias_record) do
          {:ok, _} -> send_json(conn, 200, %{ok: true})
          {:error, _} -> send_json(conn, 500, %{error: "Failed to delete"})
        end
    end
  end

  match _ do
    send_json(conn, 404, %{error: "Not found"})
  end

  # --- Serialization ---

  defp serialize_provider(provider) do
    %{
      id: provider.id,
      name: provider.name,
      api_type: to_string(provider.api_type),
      api_url: provider.api_url,
      api_key_hint: Provider.api_key_hint(provider),
      models: provider.models,
      rpm_limit: provider.rpm_limit,
      default_headers: provider.default_headers,
      enabled: provider.enabled,
      aliases: Enum.map(provider.aliases || [], &serialize_alias/1),
      inserted_at: provider.inserted_at,
      updated_at: provider.updated_at
    }
  end

  defp serialize_alias(alias_record) do
    %{
      id: alias_record.id,
      alias: alias_record.alias,
      model: alias_record.model,
      provider_id: alias_record.provider_id
    }
  end

  defp send_json(conn, status, body) do
    conn
    |> put_resp_content_type("application/json")
    |> send_resp(status, Jason.encode!(body))
  end

  defp format_errors(changeset) do
    Ecto.Changeset.traverse_errors(changeset, fn {msg, opts} ->
      Enum.reduce(opts, msg, fn {key, value}, acc ->
        String.replace(acc, "%{#{key}}", to_string(value))
      end)
    end)
  end

  defp atomize_keys(map) when is_map(map) do
    Map.new(map, fn
      {"api_type", v} -> {:api_type, String.to_existing_atom(v)}
      {k, v} when is_binary(k) -> {String.to_existing_atom(k), v}
      pair -> pair
    end)
  rescue
    ArgumentError -> map
  end
end
```

```elixir
defmodule Backplane.LLM.ModelExtractor do
  @moduledoc """
  Extracts and replaces the \"model\" field in LLM request JSON bodies.
  Uses Jason for targeted extraction — the body may be large (images, tools).
  """

  @doc "Extract the model string from a JSON body."
  @spec extract(String.t()) :: {:ok, String.t()} | {:error, :no_model | :invalid_json}
  def extract(body) when is_binary(body) do
    case Jason.decode(body) do
      {:ok, %{"model" => model}} when is_binary(model) and model != "" ->
        {:ok, model}

      {:ok, _} ->
        {:error, :no_model}

      {:error, _} ->
        {:error, :invalid_json}
    end
  end

  @doc "Replace the model field in a JSON body with a new value."
  @spec replace_model(String.t(), String.t()) :: {:ok, String.t()} | {:error, :invalid_json}
  def replace_model(body, new_model) when is_binary(body) and is_binary(new_model) do
    case Jason.decode(body) do
      {:ok, decoded} ->
        {:ok, Jason.encode!(Map.put(decoded, "model", new_model))}

      {:error, _} ->
        {:error, :invalid_json}
    end
  end
```

```elixir
# apps/backplane/lib/backplane/llm/model_resolver.ex
defmodule Backplane.LLM.ModelResolver do
  @moduledoc """
  Resolves model strings to {provider, raw_model} using prefixed names and aliases.
  Results are cached in ETS with a 30-second TTL, invalidated via PubSub.
  """

  use GenServer
  import Ecto.Query

  alias Backplane.LLM.{Provider, ModelAlias}
  alias Backplane.Repo

  @cache_table :llm_model_resolver_cache
  @ttl_ms 30_000

  # --- Client API ---

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  @doc """
  Resolve a model string to {provider, raw_model}.
  Returns {:ok, provider, raw_model} | {:error, :no_provider} | {:error, :api_type_mismatch, provider}
  """
  @spec resolve(atom(), String.t()) :: {:ok, Provider.t(), String.t()} | {:error, atom()} | {:error, atom(), Provider.t()}
  def resolve(api_type, model_string) do
    cache_key = {api_type, model_string}

    case lookup_cache(cache_key) do
      {:ok, result} -> result
      :miss -> resolve_and_cache(api_type, model_string, cache_key)
    end
  end

  @doc "Clear the entire resolution cache."
  def clear_cache do
    if :ets.info(@cache_table) != :undefined do
      :ets.delete_all_objects(@cache_table)
    end
  end

  # --- GenServer ---

  @impl true
  def init(_opts) do
    ensure_table()
    Phoenix.PubSub.subscribe(Backplane.PubSub, "llm:providers")
    {:ok, %{}}
  end

  @impl true
  def handle_info({:llm_providers_changed, _}, state) do
    clear_cache()
    {:noreply, state}
  end

  def handle_info(_, state), do: {:noreply, state}

  # --- Private ---

  defp ensure_table do
    case :ets.info(@cache_table) do
      :undefined ->
        :ets.new(@cache_table, [:set, :public, :named_table, read_concurrency: true])

      _ ->
        :ok
    end
  rescue
    ArgumentError -> :ok
  end

  defp lookup_cache(key) do
    ensure_table()

    case :ets.lookup(@cache_table, key) do
      [{^key, result, expires_at}] ->
        if System.monotonic_time(:millisecond) < expires_at do
          {:ok, result}
        else
          :ets.delete(@cache_table, key)
          :miss
        end

      [] ->
        :miss
    end
  rescue
    ArgumentError -> :miss
  end

  defp resolve_and_cache(api_type, model_string, cache_key) do
    result = do_resolve(api_type, model_string)
    expires_at = System.monotonic_time(:millisecond) + @ttl_ms

    try do
      :ets.insert(@cache_table, {cache_key, result, expires_at})
    rescue
      ArgumentError -> :ok
    end

    result
  end

  defp do_resolve(api_type, model_string) do
    case String.split(model_string, "/", parts: 2) do
      [provider_name, raw_model] ->
        resolve_prefixed(api_type, provider_name, raw_model)

      [_unprefixed] ->
        resolve_alias(api_type, model_string)
    end
  end

  defp resolve_prefixed(api_type, provider_name, raw_model) do
    query =
      from p in Provider,
        where: p.name == ^provider_name and is_nil(p.deleted_at)

    case Repo.one(query) do
      nil ->
        {:error, :no_provider}

      %{enabled: false} ->
        {:error, :no_provider}

      %{api_type: ^api_type} = provider ->
        if raw_model in provider.models do
          {:ok, provider, raw_model}
        else
          {:error, :no_provider}
        end

      provider ->
        {:error, :api_type_mismatch, provider}
    end
  end

  defp resolve_alias(api_type, alias_name) do
    query =
      from a in ModelAlias,
        join: p in Provider,
        on: a.provider_id == p.id,
        where: a.alias == ^alias_name and p.enabled == true and is_nil(p.deleted_at),
        select: {p, a.model}

    case Repo.one(query) do
      nil ->
        {:error, :no_provider}

      {%{api_type: ^api_type} = provider, model} ->
        {:ok, provider, model}

      {provider, _model} ->
        {:error, :api_type_mismatch, provider}
    end
  end
end
```

```elixir
defmodule Backplane.LLM.CredentialPlug do
  @moduledoc """
  Strips client authentication headers and injects provider API credentials.
  Also merges provider default_headers.
  """

  import Plug.Conn

  alias Backplane.LLM.Provider

  @doc "Inject provider credentials into the connection, stripping client auth."
  @spec inject(Plug.Conn.t(), Provider.t()) :: Plug.Conn.t()
  def inject(conn, %Provider{api_type: :anthropic} = provider) do
    {:ok, api_key} = Provider.decrypt_api_key(provider)

    conn
    |> delete_req_header("authorization")
    |> put_req_header("x-api-key", api_key)
    |> maybe_inject_anthropic_version()
    |> merge_default_headers(provider.default_headers)
  end

  def inject(conn, %Provider{api_type: :openai} = provider) do
    {:ok, api_key} = Provider.decrypt_api_key(provider)

    conn
    |> delete_req_header("x-api-key")
    |> put_req_header("authorization", "Bearer #{api_key}")
    |> merge_default_headers(provider.default_headers)
  end

  defp maybe_inject_anthropic_version(conn) do
    case get_req_header(conn, "anthropic-version") do
      [] -> put_req_header(conn, "anthropic-version", "2023-06-01")
      _ -> conn
    end
  end

  defp merge_default_headers(conn, nil), do: conn
  defp merge_default_headers(conn, headers) when map_size(headers) == 0, do: conn

  defp merge_default_headers(conn, headers) do
    Enum.reduce(headers, conn, fn {key, value}, acc ->
      put_req_header(acc, String.downcase(key), value)
    end)
  end
end
```

```elixir
defmodule Backplane.LLM.RouteLoader do
  @moduledoc """
  Registers and maintains Relayixir upstream configs for LLM providers.
  Subscribes to PubSub for provider changes.
  """

  use GenServer
  require Logger

  alias Backplane.LLM.Provider
  alias Relayixir.Config.UpstreamConfig

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  @impl true
  def init(_opts) do
    Phoenix.PubSub.subscribe(Backplane.PubSub, "llm:providers")
    load_all_providers()
    {:ok, %{}}
  end

  @impl true
  def handle_info({:llm_providers_changed, _}, state) do
    load_all_providers()
    {:noreply, state}
  end

  def handle_info(_, state), do: {:noreply, state}

  defp load_all_providers do
    providers = Provider.list()

    # Build new upstream map for LLM providers
    new_upstreams =
      for provider <- providers, into: %{} do
        {upstream_name(provider.id), build_upstream_config(provider)}
      end

    # Get current upstreams, filter out LLM ones, merge new
    current = UpstreamConfig.list_upstreams()

    non_llm =
      Map.reject(current, fn {name, _} -> String.starts_with?(name, "llm_provider_") end)

    UpstreamConfig.put_upstreams(Map.merge(non_llm, new_upstreams))
  rescue
    e ->
      Logger.warning("Failed to load LLM provider upstreams: #{Exception.message(e)}")
  end

  defp build_upstream_config(%Provider{api_url: api_url}) do
    uri = URI.parse(api_url)

    %{
      scheme: String.to_atom(uri.scheme || "https"),
      host: uri.host,
      port: uri.port || (if uri.scheme == "https", do: 443, else: 80),
      path_prefix_rewrite: uri.path || "",
      # LLM requests can be large and slow — generous limits
      max_request_body_size: 50_000_000,
      max_response_body_size: 50_000_000,
      request_timeout: 300_000,
      first_byte_timeout: 120_000,
      connect_timeout: 10_000
    }
  end

  @doc "Get the Relayixir upstream name for a provider."
  def upstream_name(provider_id), do: "llm_provider_#{provider_id}"
end
```

```elixir
# apps/backplane/lib/backplane/llm/router.ex
defmodule Backplane.LLM.Router do
  @moduledoc """
  Plug.Router for LLM proxy requests.
  Handles /v1/messages (Anthropic), /v1/chat/completions (OpenAI), /v1/models (aggregated listing).
  Mounted at /llm in the Phoenix router.
  """

  use Plug.Router
  require Logger

  alias Backplane.LLM.{
    ModelExtractor,
    ModelResolver,
    CredentialPlug,
    Provider,
    ModelAlias,
    RouteLoader
  }

  alias Relayixir.Config.UpstreamConfig
  alias Relayixir.Proxy.{HttpPlug, Upstream}

  plug Backplane.Transport.CORS
  plug :match
  plug Backplane.Transport.AuthPlug
  plug :dispatch

  # --- Model Listing ---

  get "/v1/models" do
    models = build_model_list()

    send_json(conn, 200, %{
      object: "list",
      data: models
    })
  end

  # --- Anthropic Messages API ---

  post "/v1/messages" do
    proxy_request(conn, :anthropic)
  end

  # --- OpenAI Chat Completions API ---

  post "/v1/chat/completions" do
    proxy_request(conn, :openai)
  end

  # --- Catch-all proxy for other POST paths ---

  post _ do
    # Determine api_type from path heuristic
    api_type =
      cond do
        String.contains?(conn.request_path, "/v1/messages") -> :anthropic
        true -> :openai
      end

    proxy_request(conn, api_type)
  end

  match _ do
    send_json(conn, 404, %{error: "Not found"})
  end

  # --- Proxy Logic ---

  defp proxy_request(conn, api_type) do
    with {:ok, body, conn} <- read_body_cached(conn),
         {:ok, model_string} <- ModelExtractor.extract(body),
         {:ok, provider, raw_model} <- resolve_model(api_type, model_string) do
      # Rewrite model field in body
      {:ok, rewritten_body} = ModelExtractor.replace_model(body, raw_model)

      # Build upstream and proxy
      upstream_name = RouteLoader.upstream_name(provider.id)

      case UpstreamConfig.get_upstream(upstream_name) do
        nil ->
          send_error(conn, api_type, 502, "Provider upstream not configured")

        upstream_config ->
          upstream = build_upstream(upstream_config, provider)

          conn
          |> CredentialPlug.inject(provider)
          |> assign(:raw_body, rewritten_body)
          |> HttpPlug.call(upstream)
      end
    else
      {:error, :invalid_json} ->
        send_error(conn, api_type, 400, "Invalid JSON request body")

      {:error, :no_model} ->
        send_error(conn, api_type, 400, "Missing 'model' field in request body")

      {:error, :no_provider} ->
        model = get_model_from_body(conn)
        send_not_found(conn, api_type, model)

      {:error, :api_type_mismatch, provider} ->
        send_api_type_mismatch(conn, api_type, provider)
    end
  end

  defp resolve_model(api_type, model_string) do
    ModelResolver.resolve(api_type, model_string)
  end

  defp read_body_cached(conn) do
    case Plug.Conn.read_body(conn) do
      {:ok, body, conn} -> {:ok, body, conn}
      {:more, _partial, _conn} -> {:error, :body_too_large}
      {:error, reason} -> {:error, reason}
    end
  end

  defp build_upstream(config, _provider) do
    %Upstream{
      scheme: Map.get(config, :scheme, :https),
      host: Map.fetch!(config, :host),
      port: Map.get(config, :port, 443),
      path_prefix_rewrite: Map.get(config, :path_prefix_rewrite),
      request_timeout: Map.get(config, :request_timeout, 300_000),
      connect_timeout: Map.get(config, :connect_timeout, 10_000),
      first_byte_timeout: Map.get(config, :first_byte_timeout, 120_000),
      max_request_body_size: Map.get(config, :max_request_body_size, 50_000_000),
      max_response_body_size: Map.get(config, :max_response_body_size, 50_000_000),
      host_forward_mode: :rewrite_to_upstream
    }
  end

  # --- Model Listing ---

  defp build_model_list do
    providers = Provider.list()

    models =
      for provider <- providers, model <- provider.models do
        %{
          id: "#{provider.name}/#{model}",
          object: "model",
          created: DateTime.to_unix(provider.inserted_at),
          owned_by: provider.name
        }
      end

    # Add alias entries
    aliases = ModelAlias.list()

    alias_models =
      for alias_record <- aliases,
          alias_record.provider != nil,
          alias_record.provider.enabled,
          is_nil(alias_record.provider.deleted_at) do
        %{
          id: alias_record.alias,
          object: "model",
          created: DateTime.to_unix(alias_record.provider.inserted_at),
          owned_by: alias_record.provider.name
        }
      end

    models ++ alias_models
  end

  # --- Error Responses ---

  defp send_not_found(conn, :anthropic, model) do
    send_json(conn, 404, %{
      type: "error",
      error: %{
        type: "not_found_error",
        message: "Model '#{model}' not found. Use {provider}/{model} or configure an alias."
      }
    })
  end

  defp send_not_found(conn, :openai, model) do
    send_json(conn, 404, %{
      error: %{
        message: "Model '#{model}' not found. Use {provider}/{model} or configure an alias.",
        type: "invalid_request_error",
        code: "model_not_found"
      }
    })
  end

  defp send_api_type_mismatch(conn, :anthropic, provider) do
    send_json(conn, 400, %{
      type: "error",
      error: %{
        type: "invalid_request_error",
        message: "Provider '#{provider.name}' is type '#{provider.api_type}', but request path requires 'anthropic'."
      }
    })
  end

  defp send_api_type_mismatch(conn, :openai, provider) do
    send_json(conn, 400, %{
      error: %{
        message: "Provider '#{provider.name}' is type '#{provider.api_type}', but request path requires 'openai'.",
        type: "invalid_request_error",
        code: "api_type_mismatch"
      }
    })
  end

  defp send_error(conn, :anthropic, status, message) do
    send_json(conn, status, %{
      type: "error",
      error: %{type: "invalid_request_error", message: message}
    })
  end

  defp send_error(conn, :openai, status, message) do
    send_json(conn, status, %{
      error: %{message: message, type: "invalid_request_error", code: "invalid_request"}
    })
  end

  defp send_error(conn, _, status, message) do
    send_json(conn, status, %{error: message})
  end

  defp send_json(conn, status, body) do
    conn
    |> put_resp_content_type("application/json")
    |> send_resp(status, Jason.encode!(body))
  end

  defp get_model_from_body(conn) do
    case conn.body_params do
      %{"model" => model} when is_binary(model) -> model
      _ -> "unknown"
    end
  rescue
    _ -> "unknown"
  end
end
```

```elixir
defmodule Backplane.Jobs.UsageRetention do
  @moduledoc "Oban cron worker that deletes old LLM usage logs."

  use Oban.Worker, queue: :llm

  import Ecto.Query
  alias Backplane.LLM.UsageLog
  alias Backplane.Repo

  @default_retention_days 90

  @impl Oban.Worker
  def perform(%Oban.Job{}) do
    retention_days = Application.get_env(:backplane, :llm_usage_retention_days, @default_retention_days)
    cutoff = DateTime.add(DateTime.utc_now(), -retention_days * 86400, :second)

    {deleted, _} =
      from(l in UsageLog, where: l.inserted_at < ^cutoff)
      |> Repo.delete_all()

    if deleted > 0 do
      require Logger
      Logger.info("LLM usage retention: deleted #{deleted} logs older than #{retention_days} days")
    end

    :ok
  end
end
```

```elixir
# apps/backplane/lib/backplane/llm/rate_limiter.ex
defmodule Backplane.LLM.RateLimiter do
  @moduledoc """
  Per-provider ETS-based sliding window rate limiter.
  Uses :counters for atomic increment and a window start timestamp.
  """

  @table :llm_rate_limiter
  @window_ms 60_000

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__.Server, opts, name: __MODULE__)
  end

  @doc "Check if a request is allowed for the given provider. Returns :ok or {:error, retry_after_seconds}."
  @spec check(String.t(), integer() | nil) :: :ok | {:error, integer()}
  def check(_provider_id, nil), do: :ok

  def check(provider_id, rpm_limit) when is_integer(rpm_limit) do
    ensure_table()
    now = System.monotonic_time(:millisecond)

    case :ets.lookup(@table, provider_id) do
      [{^provider_id, counter_ref, window_start}] ->
        if now - window_start > @window_ms do
          # Window expired — reset
          :counters.put(counter_ref, 1, 1)
          :ets.insert(@table, {provider_id, counter_ref, now})
          :ok
        else
          current = :counters.get(counter_ref, 1)

          if current < rpm_limit do
            :counters.add(counter_ref, 1, 1)
            :ok
          else
            retry_after = div(@window_ms - (now - window_start), 1000) + 1
            {:error, max(retry_after, 1)}
          end
        end

      [] ->
        counter_ref = :counters.new(1, [:atomics])
        :counters.put(counter_ref, 1, 1)
        :ets.insert(@table, {provider_id, counter_ref, now})
        :ok
    end
  end

  @doc "Reset all rate limit state."
  def reset do
    if :ets.info(@table) != :undefined do
      :ets.delete_all_objects(@table)
    end
  end

  @doc "Manually expire a provider's window (for testing)."
  def expire(provider_id) do
    case :ets.lookup(@table, provider_id) do
      [{^provider_id, counter_ref, _}] ->
        :ets.insert(@table, {provider_id, counter_ref, 0})

      [] ->
        :ok
    end
  end

  defp ensure_table do
    case :ets.info(@table) do
      :undefined ->
        try do
          :ets.new(@table, [:set, :public, :named_table, write_concurrency: true])
        rescue
          ArgumentError -> :ok
        end

      _ ->
        :ok
    end
  end

  # Simple GenServer to own the ETS table
  defmodule Server do
    use GenServer

    def init(_opts) do
      case :ets.info(:llm_rate_limiter) do
        :undefined ->
          :ets.new(:llm_rate_limiter, [:set, :public, :named_table, write_concurrency: true])

        _ ->
          :ok
      end

      {:ok, %{}}
    end
  end
end
```

```elixir
defmodule Backplane.LLM.UsageQuery do
  @moduledoc "Aggregation queries over LLM usage logs."

  import Ecto.Query
  alias Backplane.LLM.UsageLog
  alias Backplane.Repo

  @type aggregate_result :: %{
          total_requests: integer(),
          total_input_tokens: integer(),
          total_output_tokens: integer(),
          by_model: [map()],
          by_status: map(),
          avg_latency_ms: integer() | nil
        }

  @spec aggregate(map()) :: aggregate_result()
  def aggregate(filters \\ %{}) do
    base = build_base_query(filters)

    totals =
      base
      |> select([l], %{
        total_requests: count(l.id),
        total_input_tokens: coalesce(sum(l.input_tokens), 0),
        total_output_tokens: coalesce(sum(l.output_tokens), 0),
        avg_latency_ms: type(avg(l.latency_ms), :integer)
      })
      |> Repo.one()

    by_model =
      base
      |> group_by([l], l.model)
      |> select([l], %{
        model: l.model,
        requests: count(l.id),
        input_tokens: coalesce(sum(l.input_tokens), 0),
        output_tokens: coalesce(sum(l.output_tokens), 0)
      })
      |> Repo.all()

    by_status =
      base
      |> group_by([l], l.status)
      |> select([l], {l.status, count(l.id)})
      |> Repo.all()
      |> Map.new(fn {status, count} -> {to_string(status), count} end)

    %{
      total_requests: totals.total_requests || 0,
      total_input_tokens: totals.total_input_tokens || 0,
      total_output_tokens: totals.total_output_tokens || 0,
      avg_latency_ms: totals.avg_latency_ms || 0,
      by_model: by_model,
      by_status: by_status
    }
  end

  defp build_base_query(filters) do
    query = from(l in UsageLog)

    query =
      if id = filters[:provider_id] do
        where(query, [l], l.provider_id == ^id)
      else
        query
      end

    query =
      if model = filters[:model] do
        where(query, [l], l.model == ^model)
      else
        query
      end

    query =
      if since = filters[:since] do
        where(query, [l], l.inserted_at >= ^since)
      else
        query
      end

    query =
      if until_ = filters[:until] do
        where(query, [l], l.inserted_at <= ^until_)
      else
        query
      end

    query
  end
end
```

```elixir
# apps/backplane/lib/backplane/llm/health_checker.ex
defmodule Backplane.LLM.HealthChecker do
  @moduledoc """
  Periodic health probing for LLM providers.
  Stores health state in ETS. Does not affect routing.
  """

  use GenServer
  require Logger

  alias Backplane.LLM.Provider

  @table :llm_health_state
  @default_interval 60_000

  def start_link(opts \\ []) do
    GenServer.start_link(__MODULE__, opts, name: __MODULE__)
  end

  @spec healthy?(String.t()) :: boolean()
  def healthy?(provider_id) do
    case :ets.lookup(@table, provider_id) do
      [{^provider_id, healthy}] -> healthy
      [] -> false
    end
  rescue
    ArgumentError -> false
  end

  def mark_healthy(provider_id) do
    :ets.insert(@table, {provider_id, true})
  rescue
    ArgumentError -> :ok
  end

  def mark_unhealthy(provider_id) do
    :ets.insert(@table, {provider_id, false})
  rescue
    ArgumentError -> :ok
  end

  @impl true
  def init(opts) do
    :ets.new(@table, [:set, :public, :named_table, read_concurrency: true])
    interval = Keyword.get(opts, :interval, config_interval())
    schedule_probe(interval)
    {:ok, %{interval: interval}}
  end

  @impl true
  def handle_info(:probe, state) do
    probe_all_providers()
    schedule_probe(state.interval)
    {:noreply, state}
  end

  def handle_info(_, state), do: {:noreply, state}

  defp schedule_probe(interval) do
    Process.send_after(self(), :probe, interval)
  end

  defp probe_all_providers do
    providers = Provider.list()

    for provider <- providers do
      Task.start(fn -> probe_provider(provider) end)
    end
  end

  defp probe_provider(provider) do
    url = "#{provider.api_url}/v1/models"

    headers =
      case provider.api_type do
        :anthropic ->
          case Provider.decrypt_api_key(provider) do
            {:ok, key} -> [{"x-api-key", key}]
            _ -> []
          end

        :openai ->
          case Provider.decrypt_api_key(provider) do
            {:ok, key} -> [{"authorization", "Bearer #{key}"}]
            _ -> []
          end
      end

    case Req.get(url, headers: headers, receive_timeout: 10_000) do
      {:ok, %{status: 200}} ->
        mark_healthy(provider.id)

      {:ok, resp} ->
        Logger.debug("Health check failed for #{provider.name}: status #{resp.status}")
        mark_unhealthy(provider.id)

      {:error, reason} ->
        Logger.debug("Health check failed for #{provider.name}: #{inspect(reason)}")
        mark_unhealthy(provider.id)
    end
  rescue
    e ->
      Logger.warning("Health probe error for #{provider.name}: #{Exception.message(e)}")
      mark_unhealthy(provider.id)
  end

  defp config_interval do
    Application.get_env(:backplane, :llm_health_check_interval, 60) * 1000
  end
end
```
